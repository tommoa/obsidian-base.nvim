## Days since

```base
filters:
  and:
    - file.hasTag("#timed")
    - file.hasTag("#wip")
formulas:
  time_since: date(this.file.name) - date(formula.last_done_file.name)
  overdue: formula.time_since > frequency
  last_done: formula.last_done_file.asLink(formula.last_done_file.name)
  last_done_file: file.backlinks.filter(value.asFile().inFolder("Calendar")).filter(value.asFile().name <= this.file.name).map(value.asFile()).sort().slice(-1)[0]
properties:
  formula.time_since:
    displayName: Time since
  formula.overdue:
    displayName: Overdue
  formula.last_done:
    displayName: Last done
  note.frequency:
    displayName: Frequency
views:
  - type: table
    name: Activity tracker
    order:
      - file.name
      - formula.time_since
      - formula.last_done
      - frequency
      - formula.overdue
    sort:
      - property: formula.time_since
        direction: DESC
```

## Daily log

```base
filters:
  or:
    - and:
        - file.file.hasTag("#effort")
        - file.file.hasTag("#wip")
        - or:
            - completed.isEmpty()
            - completed > date(this.file.name)
        - file.ctime.date() <= date(this.file.name)
    - file.file.inFolder("Encounters")
views:
  - type: table
    name: Personal
    filters:
      not:
        - file.hasTag("#work")
        - file.hasTag("#effort/work")
    sort:
      - column: property.rank
        direction: DESC
    order:
      - file.name
  - type: table
    name: Work
    filters:
      or:
        - file.hasTag("#effort/work")
        - file.hasTag("#work")
    sort:
      - column: property.rank
        direction: DESC
    order:
      - file.name
```
